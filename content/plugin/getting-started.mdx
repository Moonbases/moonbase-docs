## Quick start

<Tabs items={['Backend plugin', 'Frontend plugin']} defaultIndex={0}>
  <Tab>
    <Steps>
      <Step>Create a file in `api/plugins`, for example `api/plugins/hello.plugin.js`.</Step>
      <Step>Export a `register` function or a plugin object.</Step>
      <Step>Restart the API (`npm run dev` from repo root).</Step>
    </Steps>

    ```js
    // api/plugins/hello.plugin.js
    module.exports = {
      name: 'hello',
      enabled: true,
      version: '0.1.0',
      register(ctx) {
        ctx.routers.api.get('/plugins/hello', (_req, res) => {
          res.json({ status_overview: 'success', status_message: 'HELLO' });
        });
      }
    };
    ```

    <Callout type="note" title="Supported formats">
      Use `.js`, `.cjs`, `.mjs`, or a folder with `index.js`.
    </Callout>
  </Tab>
  <Tab>
    <Steps>
      <Step>Create `app/plugins/hello.tsx`.</Step>
      <Step>Export a `FrontendPlugin`.</Step>
      <Step>Register it in `app/plugins/registry.ts` (import it and add it to `plugins`).</Step>
      <Step>Ensure the app runs with `PluginHost` (already wired in `app/app/layout.tsx`).</Step>
    </Steps>

    ```tsx
    // app/plugins/hello.tsx
    import type { FrontendPlugin } from '@/lib/plugins/types';

    const HelloPlugin: FrontendPlugin = {
      name: 'hello',
      enabled: true,
      version: '0.1.0',
      slots: {
        global: () => <div className="fixed bottom-4 right-4">Hello</div>
      }
    };

    export default HelloPlugin;
    ```

    ```ts
    // app/plugins/registry.ts
    import helloPlugin from './hello';

    export const plugins = [helloPlugin];
    ```

    <Callout type="info" title="Registry is explicit">
      Add your plugin import to `app/plugins/registry.ts` and include it in the exported `plugins` array. Add `version` to label plugin releases.
    </Callout>
  </Tab>
</Tabs>

## Maintenance

<Accordions type="single" collapsible>
  <Accordion title="Removing a plugin">
    - Backend: delete the plugin file from `api/plugins`.
    - Frontend: delete the plugin file from `app/plugins` and remove it from `app/plugins/registry.ts`.
  </Accordion>
  <Accordion title="Toggle without deleting">
    Add `enabled: false` to backend or frontend plugins to skip loading without deleting the file. The API logs that the plugin is disabled and does not run it.
  </Accordion>
</Accordions>
